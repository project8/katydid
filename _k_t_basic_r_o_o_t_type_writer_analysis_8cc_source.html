78  void KTBasicROOTTypeWriterAnalysis::WriteNormalizedFSDataPolar(KTDataPtr data)
79  {
80  if (! data) return;
81 
82  uint64_t sliceNumber = data->Of<KTSliceHeader>().GetSliceNumber();
83 
84  KTNormalizedFSDataPolar& fsData = data->Of<KTNormalizedFSDataPolar>();
85  unsigned nComponents = fsData.GetNComponents();
86 
87  if (! fWriter->OpenAndVerifyFile()) return;
88 
89  for (unsigned iChannel=0; iChannel<nComponents; iChannel++)
90  {
91  const KTFrequencySpectrumPolar* spectrum = fsData.GetSpectrumPolar(iChannel);
92  if (spectrum != NULL)
93  {
94  stringstream conv;
95  conv << "histNFSpolar_" << sliceNumber << "_" << iChannel;
96  string histName;
97  conv >> histName;
98  TH1D* powerSpectrum = spectrum->CreateMagnitudeHistogram(histName);
99  powerSpectrum->SetDirectory(fWriter->GetFile());
100  powerSpectrum->Write();
101  KTDEBUG(publog, "Histogram <" << histName << "> written to ROOT file");
102  }
103  }
104  return;
