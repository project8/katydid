412  uint32_t M3Header::AddStream( const std::string& aSource, uint32_t aNChannels, uint32_t aFormat,
413  uint32_t anAcqRate, uint32_t aRecSize, uint32_t aSampleSize,
414  uint32_t aDataTypeSize, uint32_t aDataFormat,
415  uint32_t aBitDepth, uint32_t aBitAlignment,
416  std::vector< unsigned >* aChanVec )
417  {
418  LDEBUG( mlog, "Adding stream " << fNStreams << " for multiple channels with record size " << aRecSize );
419  unsigned tFirstNewChannel = fNChannels;
420  for( uint32_t iNewChannel = 0; iNewChannel < aNChannels; ++iNewChannel )
421  {
422  LDEBUG( mlog, "Adding channel " << fNChannels );
423  if( aChanVec != NULL ) aChanVec->push_back( fNChannels );
424  fChannelStreams.push_back( fNStreams );
425  fChannelHeaders.push_back( M3ChannelHeader( aSource, fNChannels, anAcqRate, aRecSize, aSampleSize, aDataTypeSize, aDataFormat, aBitDepth, aBitAlignment ) );
426  ++fNChannels;
427  fChannelCoherence.resize( fNChannels ); // resize number of columns
428  for( unsigned i = 0; i < fNChannels; ++i ) // resize all rows
429  {
430  fChannelCoherence[ i ].resize( fNChannels, false );
431  }
432  fChannelCoherence.back().back() = true; // each channel is coherent with itself
433  if( fNChannels > 0 ) // this condition is necessary because if it's not the case, fNChannels-1 will be some huge number, since we're dealing with unsigned ints, so the for-loop condition won't be sufficient
434  {
435  for( unsigned i = tFirstNewChannel; i < fNChannels - 1; ++i )
436  {
437  fChannelCoherence[ fNChannels - 1 ][ i ] = true;
438  fChannelCoherence[ i ][ fNChannels - 1 ] = true;
439  }
440  }
441  }
442  fStreamHeaders.push_back( M3StreamHeader( aSource, fNStreams, aNChannels, tFirstNewChannel, aFormat, anAcqRate, aRecSize, aSampleSize, aDataTypeSize, aDataFormat, aBitDepth, aBitAlignment ) );
443  return fNStreams++;
