312  bool M3Stream::ReadRecord( int anOffset ) const
313  {
314  if( ! fIsInitialized ) Initialize();
315 
316  anOffset += fRecordsAccessed;
317  if( ( anOffset < 0 && (unsigned)abs( anOffset ) > fRecordCountInFile ) ||
318  ( anOffset > 0 && fRecordCountInFile + anOffset >= fNRecordsInFile ) ||
319  ( anOffset == 0 && fNRecordsInFile == 0 ))
320  {
321  // either requested to go back before the beginning of the file, or past the end
322  return false;
323  }
324 
325  fRecordCountInFile = fRecordCountInFile + anOffset;
326  unsigned nextAcq = fRecordIndex.at( fRecordCountInFile ).first;
327  fRecordCountInAcq = fRecordIndex.at( fRecordCountInFile ).second;
328  LDEBUG( mlog, "Going to record " << fRecordCountInFile << " -- " << nextAcq << " -- " << fRecordCountInAcq );
329 
330  try
331  {
332  bool tIsNewAcq = false;
333  if( nextAcq != fAcquisitionId || ! fRecordsAccessed )
334  {
335  // go to a new acquisition
336  tIsNewAcq = true;
337  fRecordsAccessed = true;
338  fAcquisitionId = nextAcq;
339  delete fH5CurrentAcqDataSet;
340  u32toa( fAcquisitionId, fAcqNameBuffer );
341  fH5CurrentAcqDataSet = new H5::DataSet( fH5AcqLoc->openDataSet( fAcqNameBuffer ) );
342  H5::Attribute tAttrNRIA( fH5CurrentAcqDataSet->openAttribute( "n_records" ) );
343  tAttrNRIA.read( tAttrNRIA.getDataType(), &fNRecordsInAcq );
344  }
345 
346  fDataOffset[ 0 ] = fRecordCountInAcq;
347 
348  (this->*fDoReadRecord)( tIsNewAcq );
349  }
350  catch( H5::Exception& e )
351  {
352  LERROR( mlog, "HDF5 error while reading a record:\n\t" << e.getCDetailMsg() << " (function: " << e.getFuncName() << ")" );
353  return false;
354  }
355 
356  return true;
357  }
