303 bool ReadRecordCheck( const M3Stream* aStream, int aOffset, unsigned aDataFormat )
304 {

311  switch( aDataFormat )

317  return false;

320  case sDigitizedS:
321  if( ! PrintChannelsUInt( aStream ) )
322  {
323  LERROR( mlog, "Failed to print channels (digitized)" );
324  return false;
325  }
326  break;
327  case sAnalog:
328  switch( aStream->GetSampleSize() )
329  {
330  case 1:
331  if( ! PrintChannelsFloat( aStream ) )
332  {
333  LERROR( mlog, "Problem printing channels (float)" );
334  return 0;
335  }

337  default:
338  if( ! PrintChannelsFloatComplex( aStream ) )

340  LERROR( mlog, "Problem printing channels (float-complex)" );
341  return 0;
342  }
343  break;
344  }
345  break;
346  default:
347  LERROR( mlog, "Invalid data format" );
348  return false;
349  }
350 
351  return true;
