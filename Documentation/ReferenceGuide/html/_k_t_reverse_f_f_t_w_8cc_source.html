126  bool KTReverseFFTW::InitializeFFT(KTReverseFFTW::State intendedState, unsigned timeSize)
127  {
128  if (intendedState == kNone)
129  {
130  KTERROR(fftwlog, "Cannot initialize FFT for state <" << intendedState << ">");
131  return false;
132  }
133 
134  if (timeSize == 0) timeSize = fTimeSize;
135 
136  // Set the time size whether or not it's different from fTimeSize, since the frequency size might not be
137  // right for the intended state.
138  SetTimeSizeForState(timeSize, intendedState);
139 
140  // fTransformFlag is guaranteed to be valid in the Set method.
141  KTDEBUG(fftwlog, "Transform flag: " << fTransformFlag);
142  TransformFlagMap::const_iterator iter = fTransformFlagMap.find(fTransformFlag);
143  unsigned transformFlag = iter->second;
144 
145  // allocate the input and output arrays if they're not there already
146  if (! AllocateArrays(intendedState))
147  {
148  KTERROR(fftwlog, "Unable to allocate arrays");
149  return false;
150  }
151 
152  if (fUseWisdom)
153  {
154  KTDEBUG(fftwlog, "Reading wisdom from file <" << fWisdomFilename << ">");
155  if (fftw_import_wisdom_from_filename(fWisdomFilename.c_str()) == 0)
156  {
157  KTWARN(fftwlog, "Unable to read FFTW wisdom from file <" << fWisdomFilename << ">");
158  }
159  }
160 
161  InitializeMultithreaded();
162 
163  if (intendedState == kC2R)
164  {
165  KTDEBUG(fftwlog, "Creating C2R plan: " << fTimeSize << " time bins; reverse FFT");
166  // Add FFTW_PRESERVE_INPUT so that the input array content is not destroyed during the FFT
167  fReversePlan = fftw_plan_dft_c2r_1d(fTimeSize, fInputArray, fROutputArray, transformFlag | FFTW_PRESERVE_INPUT);
168  // deleting arrays to save space
169  // output array (fROutputArray) is required; input array is not needed
170  fftw_free(fInputArray);
171  fInputArray = NULL;
172  }
173  else // intendedState == kC2C || kRasC2C
174  {
175  KTDEBUG(fftwlog, "Creating C2C plan: " << fTimeSize << " time bins; forward FFT");
176  // Add FFTW_PRESERVE_INPUT so that the input array content is not destroyed during the FFT
177  fReversePlan = fftw_plan_dft_1d(fTimeSize, fInputArray, fCOutputArray, FFTW_BACKWARD, transformFlag | FFTW_PRESERVE_INPUT);
178  // deleting arrays to save space; neither input nor output are needed
179  FreeArrays();
180  }
181 
182  if (fReversePlan != NULL)
183  {
184  fIsInitialized = true;
185  if (fUseWisdom)
186  {
187  if (fftw_export_wisdom_to_filename(fWisdomFilename.c_str()) == 0)
188  {
189  KTWARN(fftwlog, "Unable to write FFTW wisdom to file <" << fWisdomFilename << ">");
190  }
191  }
192  KTDEBUG(fftwlog, "FFTW plan created; Initialization complete.");
193  }
194  else
195  {
196  fIsInitialized = false;
197  KTERROR(fftwlog, "Unable to create the reverse FFT plan! FFT is not initialized.");;
198  return false;
199  }
200 
201  fState = intendedState;
202  return true;
203  }

205  bool KTReverseFFTW::TransformDataToReal(KTFrequencySpectrumDataFFTW& fsData)
206  {
207  if (fState != kC2R)
208  {
209  KTERROR(fftwlog, "Cannot do transform to real data in state <" << fState << ">");
210  return false;
211  }
212 
213  if (fsData.GetSpectrumFFTW(0)->size() != GetFrequencySize())
214  {
215  SetFrequencySize(fsData.GetSpectrumFFTW(0)->size());
216  InitializeForRealTDD();
217  }
218 
219  if (! fIsInitialized)
220  {
221  KTERROR(fftwlog, "FFT must be initialized before the transform is performed\n"
222  << "\tPlease initialize the FFT first, then perform the transform.");
223  return false;
224  }
225 
226  UpdateBinningCache(fsData.GetSpectrumFFTW(0)->GetFrequencyBinWidth());
227 
228  unsigned nComponents = fsData.GetNComponents();
229 
230  KTTimeSeriesData& newData = fsData.Of< KTTimeSeriesData >().SetNComponents(nComponents);
231 
232  for (unsigned iComponent = 0; iComponent < nComponents; ++iComponent)
233  {
234  const KTFrequencySpectrumFFTW* nextInput = fsData.GetSpectrumFFTW(iComponent);
235  if (nextInput == NULL)
236  {
237  KTERROR(fftwlog, "Frequency spectrum <" << iComponent << "> does not appear to be present.");
238  return false;
239  }
240 
241  KTTimeSeriesReal* nextResult = FastTransformToReal(nextInput);
242 
243  if (nextResult == NULL)
244  {
245  KTERROR(fftwlog, "One of the channels did not transform correctly.");
246  return false;
247  }
248  newData.SetTimeSeries(nextResult, iComponent);
249  }
250 
251  KTDEBUG(fftwlog, "FFT complete; " << nComponents << " component(s) transformed");
252 
253  return true;
254  }

279  void KTReverseFFTW::DoTransform(const KTFrequencySpectrumFFTW* fsIn, KTTimeSeriesReal* tsOut) const
280  {
281  fftw_execute_dft_c2r(fReversePlan, fsIn->GetData(), fROutputArray);
282  std::copy(fROutputArray, fROutputArray+fTimeSize, tsOut->begin());
